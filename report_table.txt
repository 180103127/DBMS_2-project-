CREATE TABLE "CLIENT" 
(	"ID" NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 1 CACHE 20 NOORDER  NOCYCLE  NOKEEP  NOSCALE  NOT NULL ENABLE, 
	"CLIENT_ID" NUMBER, 
	"NAME_SURNAME" VARCHAR2(100), 
	"ADDRESS" VARCHAR2(255), 
	"EMAIL" VARCHAR2(50), 
	"PHONE_NUMBER" VARCHAR2(50), 
	 PRIMARY KEY ("ID")
USING INDEX  ENABLE
);

CREATE TABLE "EMPLOYEES" 
   (	"EMPLOYEE_ID" NUMBER(*,0), 
	"FULL_NAME" VARCHAR2(100), 
	"EMAIL" VARCHAR2(100), 
	"PHONE_NUMBER" NUMBER(*,0), 
	 PRIMARY KEY ("EMPLOYEE_ID")
USING INDEX  ENABLE
);

CREATE TABLE "ORDERS" 
   (	"ORDER_ID" NUMBER NOT NULL ENABLE, 
	"CLIENT_ID" NUMBER NOT NULL ENABLE, 
	"ORDER_DATE" DATE, 
	"EMPLOYEE_ID" NUMBER, 
	"TOTAL_PRICE" NUMBER, 
	 CONSTRAINT "ORDERS_PK" PRIMARY KEY ("ORDER_ID")
USING INDEX  ENABLE
);

CREATE TABLE "ORDER_HISTORY" 
   (	"ORDER_HISTORY_ID" NUMBER(*,0), 
	"ORDER_ID" NUMBER(*,0), 
	"PRODUCT_ID" NUMBER(*,0), 
	"QUANTITY" NUMBER(*,0), 
	"PRICE" NUMBER(*,0), 
	 PRIMARY KEY ("ORDER_HISTORY_ID")
USING INDEX  ENABLE
);
  ALTER TABLE "ORDER_HISTORY" ADD FOREIGN KEY ("ORDER_ID")
	  REFERENCES "ORDERS" ("ORDER_ID") ENABLE;
  ALTER TABLE "ORDER_HISTORY" ADD FOREIGN KEY ("PRODUCT_ID")
	  REFERENCES "PRODUCT" ("PRODUCT_ID") ENABLE;


CREATE TABLE "PAYMENT" 
   (	"PAYMENT_ID" NUMBER(*,0), 
	"NAME" VARCHAR2(50)
);

CREATE TABLE "PRICE_CHANGE" 
   (	"CHANGE_ID" NUMBER(*,0), 
	"PRODUCT_ID" NUMBER(*,0), 
	"OLD_PRICE" NUMBER(*,0), 
	"NEW_PRICE" NUMBER(*,0), 
	"CHANGE_DATE" DATE, 
	 PRIMARY KEY ("CHANGE_ID")
USING INDEX  ENABLE
);
  ALTER TABLE "PRICE_CHANGE" ADD FOREIGN KEY ("PRODUCT_ID")
	  REFERENCES "PRODUCT" ("PRODUCT_ID") ENABLE;

CREATE TABLE "PRODUCT" 
   (	"PRODUCT_ID" NUMBER(*,0), 
	"CATEGOTY_ID" NUMBER(*,0), 
	"PRICE" NUMBER(*,0), 
	"SUPLIER" VARCHAR2(50), 
	"PRODUCT_NAME" VARCHAR2(50), 
	"PRODUCT_DESCRIPTION" VARCHAR2(50), 
	 PRIMARY KEY ("PRODUCT_ID")
USING INDEX  ENABLE
);

CREATE TABLE "SALARY" 
   (	"EMPLOYEE_ID" NUMBER(*,0), 
	"START_DATE" DATE, 
	"END_DATE" DATE, 
	"SALARY" NUMBER(*,0), 
	 PRIMARY KEY ("EMPLOYEE_ID")
USING INDEX  ENABLE
);
  ALTER TABLE "SALARY" ADD FOREIGN KEY ("EMPLOYEE_ID")
	  REFERENCES "EMPLOYEES" ("EMPLOYEE_ID") ENABLE;

CREATE TABLE "STORAGE" 
   (	"QUANTITY" NUMBER(*,0), 
	"PRODUCT_NAME" VARCHAR2(50)
);

CREATE TABLE "SUPPLIER" 
   (	"SUPPLIER_ID" NUMBER(*,0), 
	"SUPPLIER_NAME" VARCHAR2(50), 
	"SUPPLIER_EMAIL" VARCHAR2(50), 
	"SUPPLIER_PHONE" NUMBER(*,0), 
	 PRIMARY KEY ("SUPPLIER_ID")
USING INDEX  ENABLE
);
