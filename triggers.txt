  CREATE OR REPLACE EDITIONABLE TRIGGER "COUNT" 
BEFORE INSERT ON CLIENT
FOR EACH ROW
DECLARE
  RowCount NUMBER;
BEGIN
  SELECT COUNT(*) INTO RowCount FROM CLIENT;
  DBMS_OUTPUT.PUT_LINE('The table has' || RowCount || ' rows.');
END;



/

  CREATE OR REPLACE EDITIONABLE TRIGGER "BI_ORDERS" 
  before insert on "ORDERS"               
  for each row  
begin   
  if :NEW."ORDER_ID" is null then 
    select "ORDERS_SEQ".nextval into :NEW."ORDER_ID" from sys.dual; 
  end if; 
end; 

/
ALTER TRIGGER "BI_ORDERS" ENABLE;


  CREATE OR REPLACE EDITIONABLE TRIGGER "ORDERS_DUPLICATE_CHECK" 
BEFORE INSERT ON orders
FOR EACH ROW
DECLARE
  v_count NUMBER;
BEGIN
  SELECT COUNT(*)
  INTO v_count
  FROM orders
  WHERE order_id = :NEW.order_id;
  IF v_count > 0 THEN
   RAISE_APPLICATION_ERROR(-00001, 'This product already exist');
  END IF;
END;
/
ALTER TRIGGER "ORDERS_DUPLICATE_CHECK" ENABLE;
